בודקת חוזק סיסמאות
כתבי פונקציית Python שמקבלת סיסמה ובודקת אם היא מספיק חזקה — שימוש מעשי בסייבר יומיומי.
  • צרי קובץ Python חדש בשם password_checker.py
  • כתבי פונקציה שמקבלת סיסמה כארגומנט
  • בדקי: אורך מינימום 8 תווים
  • בדקי: יש לפחות אות גדולה אחת (isupper())
  • בדקי: יש לפחות ספרה אחת (isdigit())
  • החזירי ציון 0–3 בהתאם לחוזק + הסבר
  • בונוס: הוסיפי בדיקה לתווים מיוחדים
# בודקת חוזק סיסמאות - Women Who Hack 🎀

def check_password(password):
    score = 0
    tips = []

    if len(password) >= 8:
        score += 1
    else:
        tips.append("❌ לפחות 8 תווים")

    if any(c.isupper() for c in password):
        score += 1
    else:
        tips.append("❌ לפחות אות גדולה אחת")

    if any(c.isdigit() for c in password):
        score += 1
    else:
        tips.append("❌ לפחות ספרה אחת")

    levels = ["חלשה 😬", "בינונית 🤔", "חזקה 💪", "מאובטחת 🔒"]
    print(f"חוזק: {levels[score]}")
    if tips:
        print("טיפים לשיפור:")
        for t in tips: print(f"  {t}")
    return score

# בדיקה
check_password("abc")
check_password("Hacker123!")

# מזהה Phishing - Women Who Hack 🎀

emails = [
    "support@paypal.com",
    "security@paypa1.com",
    "noreply@amazon.com",
    "winner@free-prize.xyz",
    "info@bank-secure.tk",
]

suspicious_domains = [".xyz", ".tk", ".ml", ".ga"]
suspicious_words = ["paypa1", "arnazon", "g00gle"]

for email in emails:
    flags = []
    for d in suspicious_domains:
        if email.endswith(d): flags.append(f"דומיין חשוד: {d}")
    for w in suspicious_words:
        if w in email: flags.append(f"שם מזויף: {w}")

    if flags:
        print(f"⚠️  {email} → {', '.join(flags)}")
    else:
        print(f"✅ {email} — נראה תקין")

סורקת פורטים פשוטה
בני כלי שסורק פורטים נפוצים בכתובת IP ומציג אילו מהם פתוחים — בדיוק כמו Nmap!
  • ייבאי את ספריית socket
  • הגדירי רשימת פורטים נפוצים: 80, 443, 22, 21, 3306
  • נסי להתחבר לכל פורט עם connect_ex()
  • אם מחזיר 0 — הפורט פתוח!
  • הגדירי timeout של שנייה אחת כדי לא לחכות יותר מדי
  • בונוס: הוסיפי מילון שממפה פורט לשם השירות
# סורקת פורטים - Women Who Hack 🎀
import socket

target = "127.0.0.1"  # בדיקה על localhost
ports = {22: "SSH", 80: "HTTP", 443: "HTTPS", 21: "FTP", 3306: "MySQL"}

print(f"🔍 סורקת {target}...\n")

for port, service in ports.items():
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(1)
    result = sock.connect_ex((target, port))
    if result == 0:
        print(f"✅ פורט {port} ({service}) — פתוח!")
    else:
        print(f"❌ פורט {port} ({service}) — סגור")
    sock.close()

print("\n🎀 סריקה הושלמה!")

Brute Force לימודי
הבני סימולציה של מתקפת Brute Force — ותגלי כמה מהר מחשב פורץ סיסמאות קצרות.
  • ייבאי itertools ו-string
  • הגדירי סיסמא סודית קצרה (2-3 תווים)
  • השתמשי ב-itertools.product לכל הצירופים האפשריים
  • ספרי כמה ניסיונות עד שמצאת
  • מדדי את הזמן עם time
  • הסיקי מסקנות: למה סיסמאות ארוכות חשובות?
# Brute Force לימודי - Women Who Hack 🎀
import itertools
import string
import time

secret = "ab3"
chars = string.ascii_lowercase + string.digits
attempts = 0
start = time.time()

print(f"🔓 מנסה לפצח סיסמא של {len(secret)} תווים...\n")

for length in range(1, len(secret) + 2):
    for combo in itertools.product(chars, repeat=length):
        attempts += 1
        guess = "".join(combo)
        if guess == secret:
            elapsed = round(time.time() - start, 3)
            print(f"✅ נמצאה! סיסמא: '{guess}'")
            print(f"⏱️  {attempts:,} ניסיונות תוך {elapsed} שניות")
            print("💡 סיסמא ארוכה = מיליארדי ניסיונות נוספים! 🎀")
            exit()

מערכת זיהוי חדירות (IDS)
כתבי סקריפט שמנטר קבצי לוג ומתריע כשיש כניסות כושלות חוזרות — בדיוק כמו מערכות IDS אמיתיות.
  • צרי קובץ לוג מדומה auth.log עם שורות "Failed login from IP"
  • קראי את הקובץ עם Python וחלצי כתובות IP
  • השתמשי ב-collections.Counter לספור כשלונות לכל IP
  • אם IP מופיע יותר מ-5 פעמים — הדפיסי "🚨 חשד ל-Brute Force!"
  • שמרי את הרשימה השחורה לקובץ blacklist.txt
  • בונוס: הוסיפי תאריך ושעה לכל התראה עם datetime
# מערכת IDS - Women Who Hack 🎀
import re
from collections import Counter
from datetime import datetime

fake_log = """
Failed login from 192.168.1.10
Failed login from 10.0.0.5
Failed login from 192.168.1.10
Failed login from 192.168.1.10
Failed login from 10.0.0.5
Failed login from 192.168.1.10
Failed login from 192.168.1.10
Failed login from 192.168.1.10
Successful login from 10.0.0.1
"""

THRESHOLD = 5
pattern = re.compile(r'Failed login from (\d+\.\d+\.\d+\.\d+)')
ips = pattern.findall(fake_log)
counts = Counter(ips)
blacklist = []

print("🔍 מנתחת לוגים...\n")
for ip, count in counts.items():
    if count >= THRESHOLD:
        now = datetime.now().strftime("%H:%M:%S")
        print(f"🚨 [{now}] חשד ל-Brute Force! IP: {ip} ({count} ניסיונות)")
        blacklist.append(ip)

with open("blacklist.txt", "w") as f:
    f.write("\n".join(blacklist))

print(f"\n🎀 {len(blacklist)} כתובות IP נוספו לרשימה השחורה!")

SQL Injection – הדגמה
הבני בסיס נתונים מדומה וכתבי קוד שמדגים איך SQL Injection עובד — ואיך מגינים מפניה.
  • ייבאי sqlite3 וצרי DB מדומה עם טבלת משתמשים
  • כתבי פונקציית login פגיעה שמרכיבה SQL ישירות מהקלט
  • הדגימי את המתקפה עם הקלט: ' OR '1'='1
  • הראי שהמתקפה עובדת — המשתמשת מתחברת ללא סיסמא!
  • כתבי פונקציית login מאובטחת עם Parameterized Queries
  • הראי שעכשיו המתקפה נכשלת ותסבירי למה
# SQL Injection הדגמה - Women Who Hack 🎀
import sqlite3

conn = sqlite3.connect(":memory:")
conn.execute("CREATE TABLE users (name TEXT, password TEXT)")
conn.execute("INSERT INTO users VALUES ('admin', 'secret123')")

# ❌ פגיעה
def login_vulnerable(user, pwd):
    q = f"SELECT * FROM users WHERE name='{user}' AND password='{pwd}'"
    return "✅ התחברת!" if conn.execute(q).fetchone() else "❌ שגיאה"

# ✅ מאובטחת
def login_safe(user, pwd):
    q = "SELECT * FROM users WHERE name=? AND password=?"
    return "✅ התחברת!" if conn.execute(q, (user, pwd)).fetchone() else "❌ שגיאה"

attack = "' OR '1'='1"
print(f"פגיע:    {login_vulnerable(attack, attack)}")
print(f"מאובטח: {login_safe(attack, attack)}")
print("\n🎀 Parameterized Queries מציל את היום!")
