בודקת חוזק סיסמא
כתבי תוכנית Python שבודקת אם סיסמא חזקה מספיק – לפי אורך, אותיות גדולות וספרות.
  • פתחי קובץ Python חדש בשם password_checker.py
  • בקשי מהמשתמשת להזין סיסמא עם input()
  • בדקי: האם האורך מעל 8 תווים?
  • בדקי: האם יש לפחות אות גדולה אחת? (השתמשי ב-any(c.isupper()))
  • בדקי: האם יש לפחות ספרה אחת? (השתמשי ב-any(c.isdigit()))
  • הדפיסי: "סיסמא חזקה ✅" או "סיסמא חלשה ❌" בהתאם לתוצאות
# בודקת חוזק סיסמא - Women Who Hack 🎀

def check_password(password):
    if len(password) < 8:
        return "❌ סיסמא קצרה מדי (מינימום 8 תווים)"
    if not any(c.isupper() for c in password):
        return "❌ חסרה אות גדולה"
    if not any(c.isdigit() for c in password):
        return "❌ חסרה ספרה"
    return "✅ סיסמא חזקה! כל הכבוד 🎀"

password = input("הכניסי סיסמא: ")
print(check_password(password))

סורקת פורטים
כתבי סקריפט Python שסורק פורטים פתוחים בכתובת IP – כלי בסיסי שכל מגנת סייבר צריכה.
  • ייבאי את המודול socket מ-Python
  • בקשי מהמשתמשת להזין כתובת IP (השתמשי ב-127.0.0.1 לבדיקה)
  • כתבי לולאה שעוברת על פורטים 1 עד 1024
  • לכל פורט – נסי לחבר בעזרת socket.connect_ex()
  • אם הפונקציה מחזירה 0 – הפורט פתוח! הדפיסי אותו
  • הוסיפי try/except לטיפול בשגיאות
# סורקת פורטים - Women Who Hack 🎀
import socket

target = input("הכניסי IP לסריקה (לדוגמא 127.0.0.1): ")
print(f"\n🔍 סורקת פורטים ב-{target}...\n")

open_ports = []

for port in range(1, 1025):
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(0.5)
        result = sock.connect_ex((target, port))
        if result == 0:
            print(f"✅ פורט {port} - פתוח!")
            open_ports.append(port)
        sock.close()
    except:
        pass

print(f"\n🎀 סריקה הושלמה! נמצאו {len(open_ports)} פורטים פתוחים")

Brute Force על סיסמא
כתבי סקריפט שמנסה לפצח סיסמא על ידי ניסיון של כל הצירופים – ותראי למה סיסמא ארוכה כל כך חשובה.
  • ייבאי itertools ו-string מ-Python
  • הגדירי סיסמא קצרה (3 תווים) לבדיקה – לדוגמא "abc"
  • כתבי לולאה עם itertools.product שמנסה כל צירוף
  • ספרי כמה ניסיונות לקח למצוא את הסיסמא
  • הדפיסי את התוצאה: "נמצאה אחרי X ניסיונות!"
  • נסי עם סיסמא של 4 תווים – שימי לב כמה זמן לוקח יותר!
# Brute Force לימודי - Women Who Hack 🎀
import itertools
import string

secret = "abc"  # שני הסיסמא כאן
chars = string.ascii_lowercase
attempts = 0

print(f"🔓 מנסה לפצח סיסמא של {len(secret)} תווים...\n")

for length in range(1, len(secret) + 2):
    for combo in itertools.product(chars, repeat=length):
        attempts += 1
        guess = "".join(combo)
        if guess == secret:
            print(f"✅ נמצאה! הסיסמא: '{guess}'")
            print(f"⏱️ לקח {attempts:,} ניסיונות")
            print("💡 לכן סיסמא ארוכה = הרבה יותר בטוחה! 🎀")
            exit()

מערכת זיהוי חדירות (IDS)
כתבי סקריפט Python שמנטר קבצי לוג ומתריע כשיש כניסות כושלות חוזרות – בדיוק כמו מערכות IDS אמיתיות.
  • צרי קובץ לוג מדומה: auth.log עם שורות של "Failed login from IP"
  • קראי את הקובץ עם Python וחלצי כתובות IP
  • השתמשי ב-collections.Counter לספור כמה פעמים כל IP נכשל
  • אם IP מופיע יותר מ-5 פעמים – הדפיסי התראה: "🚨 חשד ל-Brute Force!"
  • שמרי את הרשימה השחורה לקובץ blacklist.txt
  • בונוס: הוסיפי תאריך ושעה לכל התראה עם datetime
# מערכת IDS - Women Who Hack 🎀
import re
from collections import Counter
from datetime import datetime

# יצירת לוג מדומה
fake_log = """
Failed login from 192.168.1.10
Failed login from 10.0.0.5
Failed login from 192.168.1.10
Failed login from 192.168.1.10
Failed login from 10.0.0.5
Failed login from 192.168.1.10
Failed login from 192.168.1.10
Failed login from 192.168.1.10
Successful login from 10.0.0.1
"""

THRESHOLD = 5
pattern = re.compile(r'Failed login from (\d+\.\d+\.\d+\.\d+)')
ips = pattern.findall(fake_log)
counts = Counter(ips)
blacklist = []

print("🔍 מנתחת לוגים...\n")
for ip, count in counts.items():
    if count >= THRESHOLD:
        now = datetime.now().strftime("%H:%M:%S")
        print(f"🚨 [{now}] חשד ל-Brute Force! IP: {ip} ({count} ניסיונות)")
        blacklist.append(ip)

with open("blacklist.txt", "w") as f:
    f.write("\n".join(blacklist))

print(f"\n🎀 {len(blacklist)} כתובות IP נוספו לרשימה השחורה!")

SQL Injection – הדגמה
הבני בסיס נתונים מדומה וכתבי קוד שמדגים איך SQL Injection עובד – ואיך מגינים מפניה.
  • ייבאי sqlite3 וצרי DB מדומה עם טבלת משתמשים
  • כתבי פונקציית login פגיעה שמרכיבה SQL ישירות מהקלט
  • הדגימי את המתקפה עם הקלט: ' OR '1'='1
  • הראי שהמתקפה עובדת – המשתמשת מתחברת ללא סיסמא!
  • כתבי פונקציית login מאובטחת עם Parameterized Queries
  • הראי שעכשיו המתקפה נכשלת – ותסבירי למה
# SQL Injection הדגמה - Women Who Hack 🎀
import sqlite3

conn = sqlite3.connect(":memory:")
conn.execute("CREATE TABLE users (name TEXT, password TEXT)")
conn.execute("INSERT INTO users VALUES ('admin', 'secret123')")

# ❌ פונקציה פגיעה
def login_vulnerable(user, pwd):
    query = f"SELECT * FROM users WHERE name='{user}' AND password='{pwd}'"
    result = conn.execute(query).fetchone()
    return "✅ התחברת!" if result else "❌ שגיאה"

# ✅ פונקציה מאובטחת
def login_safe(user, pwd):
    query = "SELECT * FROM users WHERE name=? AND password=?"
    result = conn.execute(query, (user, pwd)).fetchone()
    return "✅ התחברת!" if result else "❌ שגיאה"

attack = "' OR '1'='1"

print("=== מתקפת SQL Injection ===")
print(f"פגיע:    {login_vulnerable(attack, attack)}")
print(f"מאובטח: {login_safe(attack, attack)}")
print("\n🎀 Parameterized Queries מציל את היום!")
